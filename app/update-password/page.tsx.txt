"use client";

import { useState } from "react";
import Link from "next/link";
import { createBrowserClient } from "@supabase/ssr";

export default function UpdatePasswordPage() {
  const supabase = createBrowserClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
  );

  const [password, setPassword] = useState("");
  const [msg, setMsg] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);

  async function onUpdate(e: React.FormEvent) {
    e.preventDefault();
    setMsg(null);
    setLoading(true);

    const { error } = await supabase.auth.updateUser({ password });

    setLoading(false);
    setMsg(error ? error.message : "Senha atualizada! Fa√ßa login.");
  }

  return (
    <div className="container">
      <div className="card" style={{ padding: 18, maxWidth: 540, margin: "70px auto" }}>
        <div className="row" style={{ alignItems: "center", justifyContent: "space-between" }}>
          <h1 className="h1">Nova senha</h1>
          <Link className="pill" href="/login">Login</Link>
        </div>

        <p className="small" style={{ marginTop: 10 }}>Defina sua nova senha.</p>

        <form onSubmit={onUpdate} style={{ marginTop: 14 }}>
          <input
            className="input"
            placeholder="Nova senha"
            type="password"
            value={password}
            onChange={(e) => setPassword(e.target.value)}
            required
            minLength={6}
          />
          <div style={{ height: 12 }} />
          <button className="pill pillPrimary" type="submit" disabled={loading}>
            {loading ? "Salvando..." : "Atualizar senha"}
          </button>
        </form>

        {msg && <div className="small" style={{ marginTop: 12 }}>{msg}</div>}
      </div>
    </div>
  );
}
