"use client";

import { useState } from "react";
import Link from "next/link";
import { createBrowserClient } from "@supabase/ssr";

export default function ResetPasswordPage() {
  const supabase = createBrowserClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
  );

  const [email, setEmail] = useState("");
  const [msg, setMsg] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);

  async function onSend(e: React.FormEvent) {
    e.preventDefault();
    setMsg(null);
    setLoading(true);

    const { error } = await supabase.auth.resetPasswordForEmail(email, {
      redirectTo: `${window.location.origin}/update-password`,
    });

    setLoading(false);
    setMsg(error ? error.message : "Email enviado! Verifique sua caixa de entrada.");
  }

  return (
    <div className="container">
      <div className="card" style={{ padding: 18, maxWidth: 540, margin: "70px auto" }}>
        <div className="row" style={{ alignItems: "center", justifyContent: "space-between" }}>
          <h1 className="h1">Recuperar senha</h1>
          <Link className="pill" href="/login">Voltar</Link>
        </div>

        <p className="small" style={{ marginTop: 10 }}>
          Informe seu e-mail para receber o link de redefinição.
        </p>

        <form onSubmit={onSend} style={{ marginTop: 14 }}>
          <input
            className="input"
            placeholder="E-mail"
            value={email}
            onChange={(e) => setEmail(e.target.value)}
            required
          />
          <div style={{ height: 12 }} />
          <button className="pill pillPrimary" type="submit" disabled={loading}>
            {loading ? "Enviando..." : "Enviar link"}
          </button>
        </form>

        {msg && <div className="small" style={{ marginTop: 12 }}>{msg}</div>}
      </div>
    </div>
  );
}
